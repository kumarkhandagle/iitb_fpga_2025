`timescale 1ns / 1ps

module spi_dac_tb;

  // Inputs
  reg sys_clk;
  reg rst_n;
  reg [11:0] dac_in1;
  reg start;

  // Outputs
  wire mosi1;
  wire sclk;
  wire cs_n;
  wire done;

  // Instantiate the Unit Under Test (UUT)
  spi_dac uut (
    .sys_clk(sys_clk),
    .rst_n(rst_n),
    .mosi1(mosi1),
    .sclk(sclk),
    .cs_n(cs_n),
    .dac_in1(dac_in1),
    .start(start),
    .done(done)
  );

  // Clock generation (50 MHz)
  initial sys_clk = 0;
  always #10 sys_clk = ~sys_clk;  // 20 ns period = 50 MHz

  // Stimulus
  initial begin

    // Initial values
    rst_n = 0;
    start = 0;
    dac_in1 = 12'd0;

    // Apply reset
    #100;
    rst_n = 1;

    // Wait for few clock cycles
    @(posedge sclk);

    // Apply input data and assert start
    dac_in1 = 12'hABC;
    start   = 1;

    @(posedge sclk);
    start = 0;

   @(posedge done);

    #100;
    $finish;
  end



endmodule
